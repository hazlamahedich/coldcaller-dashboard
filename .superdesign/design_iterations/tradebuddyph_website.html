import backtrader as bt
import pandas as pd

class MomentumStrategy(bt.Strategy):
    def __init__(self):
        self.ma20 = bt.indicators.SMA(period=20)
        self.ma50 = bt.indicators.SMA(period=50)
        self.rsi = bt.indicators.RSI(period=14)
        
    def next(self):
        if not self.position:
            if (self.ma20[0] > self.ma50[0] and 
                self.rsi[0] < 70):
                self.buy(size=100)
                
        elif self.position:
            if (self.data.close[0] <= self.position.price * 0.98 or
                self.data.close[0] >= self.position.price * 1.05):
                self.sell(size=self.position.size)
